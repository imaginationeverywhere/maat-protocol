# Short Video Standard

## Overview
Short-form video processing and delivery for TikTok-style vertical videos. Handles upload, transcoding, thumbnail generation, audio extraction, and optimized streaming delivery.

## Domain Context
- **Primary Projects**: Quik Social, Quik Music (music videos), Quik Events (event clips)
- **Related Domains**: Social Feed, Live Streaming, Music
- **Key Integration**: FFmpeg, AWS MediaConvert, Cloudflare Stream, Mux

## Core Interfaces

```typescript
interface ShortVideo {
  id: string;
  authorId: string;
  status: VideoStatus;
  source: VideoSource;
  processing: ProcessingStatus;
  metadata: VideoMetadata;
  streams: VideoStream[];
  thumbnail: VideoThumbnail;
  audio?: ExtractedAudio;
  captions?: VideoCaptions;
  analytics: VideoAnalytics;
  createdAt: Date;
  publishedAt?: Date;
}

type VideoStatus = 'uploading' | 'processing' | 'ready' | 'failed' | 'deleted';

interface VideoSource {
  originalUrl: string;
  fileSize: number;
  mimeType: string;
  uploadedAt: Date;
}

interface ProcessingStatus {
  stage: ProcessingStage;
  progress: number;
  startedAt?: Date;
  completedAt?: Date;
  error?: string;
}

type ProcessingStage =
  | 'queued'
  | 'downloading'
  | 'analyzing'
  | 'transcoding'
  | 'generating_thumbnails'
  | 'extracting_audio'
  | 'generating_captions'
  | 'uploading_outputs'
  | 'completed'
  | 'failed';

interface VideoMetadata {
  duration: number;
  width: number;
  height: number;
  aspectRatio: string;
  orientation: 'portrait' | 'landscape' | 'square';
  frameRate: number;
  bitrate: number;
  codec: string;
  hasAudio: boolean;
  audioCodec?: string;
}

interface VideoStream {
  quality: VideoQuality;
  url: string;
  width: number;
  height: number;
  bitrate: number;
  codec: string;
}

type VideoQuality = '360p' | '480p' | '720p' | '1080p';

interface VideoThumbnail {
  url: string;
  animatedUrl?: string;
  blurhash: string;
  sprites?: ThumbnailSprite[];
  customTimestamp?: number;
}

interface ThumbnailSprite {
  url: string;
  interval: number;
  columns: number;
  rows: number;
}

interface ExtractedAudio {
  id: string;
  url: string;
  duration: number;
  waveform: number[];
  isOriginal: boolean;
  linkedSongId?: string;
  title?: string;
  artist?: string;
}

interface VideoCaptions {
  language: string;
  url: string;
  format: 'vtt' | 'srt';
  autoGenerated: boolean;
  segments: CaptionSegment[];
}

interface CaptionSegment {
  startTime: number;
  endTime: number;
  text: string;
  confidence?: number;
}

interface VideoAnalytics {
  views: number;
  uniqueViews: number;
  completionRate: number;
  averageWatchTime: number;
  engagementScore: number;
  shares: number;
  saves: number;
}

interface VideoUploadRequest {
  file: File | Blob;
  caption?: string;
  audioId?: string;
  effects?: VideoEffect[];
  trimStart?: number;
  trimEnd?: number;
  coverTimestamp?: number;
}

interface VideoEffect {
  type: EffectType;
  parameters: Record<string, any>;
  startTime?: number;
  endTime?: number;
}

type EffectType =
  | 'filter'
  | 'speed'
  | 'reverse'
  | 'text_overlay'
  | 'sticker'
  | 'transition'
  | 'green_screen';
```

## Processing Pipeline

```typescript
interface ProcessingPipeline {
  // 1. Upload handling
  initiateUpload(request: VideoUploadRequest): Promise<UploadSession>;
  completeUpload(sessionId: string): Promise<ShortVideo>;

  // 2. Analysis
  analyzeVideo(videoId: string): Promise<VideoMetadata>;
  detectContent(videoId: string): Promise<ContentAnalysis>;

  // 3. Transcoding
  transcodeToQualities(videoId: string, qualities: VideoQuality[]): Promise<VideoStream[]>;
  applyEffects(videoId: string, effects: VideoEffect[]): Promise<void>;

  // 4. Asset generation
  generateThumbnails(videoId: string): Promise<VideoThumbnail>;
  extractAudio(videoId: string): Promise<ExtractedAudio>;
  generateCaptions(videoId: string, language: string): Promise<VideoCaptions>;

  // 5. Delivery optimization
  uploadToCDN(videoId: string): Promise<void>;
  generateManifest(videoId: string): Promise<string>;
}

interface ContentAnalysis {
  nsfw: boolean;
  nsfwScore: number;
  copyrightMusic: boolean;
  detectedSong?: DetectedSong;
  scenes: SceneDetection[];
  objects: ObjectDetection[];
  text: TextDetection[];
}

interface DetectedSong {
  songId: string;
  title: string;
  artist: string;
  confidence: number;
  startTime: number;
  endTime: number;
}
```

## Database Schema

```sql
CREATE TABLE short_videos (
  id UUID PRIMARY KEY,
  author_id UUID NOT NULL REFERENCES users(id),
  status VARCHAR(20) DEFAULT 'uploading',
  original_url TEXT NOT NULL,
  file_size BIGINT NOT NULL,
  mime_type VARCHAR(50),
  duration DECIMAL(10,3),
  width INTEGER,
  height INTEGER,
  aspect_ratio VARCHAR(10),
  orientation VARCHAR(20),
  frame_rate DECIMAL(5,2),
  bitrate INTEGER,
  codec VARCHAR(20),
  has_audio BOOLEAN DEFAULT true,
  audio_codec VARCHAR(20),
  processing_stage VARCHAR(30) DEFAULT 'queued',
  processing_progress INTEGER DEFAULT 0,
  processing_error TEXT,
  thumbnail_url TEXT,
  animated_thumbnail_url TEXT,
  blurhash VARCHAR(100),
  view_count INTEGER DEFAULT 0,
  unique_view_count INTEGER DEFAULT 0,
  published_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE video_streams (
  id UUID PRIMARY KEY,
  video_id UUID NOT NULL REFERENCES short_videos(id),
  quality VARCHAR(10) NOT NULL,
  url TEXT NOT NULL,
  width INTEGER NOT NULL,
  height INTEGER NOT NULL,
  bitrate INTEGER NOT NULL,
  codec VARCHAR(20),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE video_audio_tracks (
  id UUID PRIMARY KEY,
  video_id UUID NOT NULL REFERENCES short_videos(id),
  url TEXT NOT NULL,
  duration DECIMAL(10,3),
  waveform JSONB,
  is_original BOOLEAN DEFAULT true,
  linked_song_id UUID,
  title VARCHAR(255),
  artist VARCHAR(255),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE video_captions (
  id UUID PRIMARY KEY,
  video_id UUID NOT NULL REFERENCES short_videos(id),
  language VARCHAR(10) NOT NULL,
  url TEXT NOT NULL,
  format VARCHAR(10) DEFAULT 'vtt',
  auto_generated BOOLEAN DEFAULT true,
  segments JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE video_views (
  id UUID PRIMARY KEY,
  video_id UUID NOT NULL REFERENCES short_videos(id),
  viewer_id UUID REFERENCES users(id),
  session_id VARCHAR(100),
  watch_duration DECIMAL(10,3),
  completion_rate DECIMAL(5,4),
  source VARCHAR(30),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_videos_author ON short_videos(author_id);
CREATE INDEX idx_videos_status ON short_videos(status);
CREATE INDEX idx_videos_published ON short_videos(published_at DESC);
CREATE INDEX idx_video_streams_video ON video_streams(video_id);
CREATE INDEX idx_video_views_video ON video_views(video_id);
CREATE INDEX idx_video_views_created ON video_views(created_at);
```

## FFmpeg Commands

```bash
# Transcode to 720p HLS
ffmpeg -i input.mp4 \
  -vf "scale=720:-2" \
  -c:v libx264 -preset fast -crf 22 \
  -c:a aac -b:a 128k \
  -hls_time 4 -hls_list_size 0 \
  -f hls output_720p.m3u8

# Generate animated thumbnail (GIF)
ffmpeg -i input.mp4 \
  -vf "fps=10,scale=320:-1:flags=lanczos" \
  -t 3 -loop 0 \
  thumbnail.gif

# Extract audio
ffmpeg -i input.mp4 \
  -vn -c:a aac -b:a 192k \
  audio.m4a

# Generate sprite sheet for scrubbing
ffmpeg -i input.mp4 \
  -vf "fps=1,scale=160:-1,tile=10x10" \
  -frames:v 1 sprites.jpg
```
